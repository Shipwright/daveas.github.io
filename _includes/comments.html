Parse.initialize("APP_ID", "JS_KEY"); var Contactmessage = Parse.Object.extend("Contactmessage"); $(document).ready(function(){ $("#contactform").on('submit',function(e){ e.preventDefault(); var data = {}; data.name = $('#name').val(); data.email = $('#email').val(); data.message = $('#message').val(); var contact = new Contactmessage(); contact.save(data, { success: function(){ $('input[type="submit"]').addClass('success'); $('input[type="submit"]').after("Success!"); $('#contactform')[0].reset(); }, error: function() { $('input[type="submit"]').css('error'); $('input[type="submit"]').after("Error!"); } }); }); });
